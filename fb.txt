where RDB$RELATION_NAME in (select RDB$RELATION_NAME from RDB$RELATIONS
where (RDB$SYSTEM_FLAG = 0) AND (RDB$RELATION_TYPE = 0)
order by RDB$RELATION_NAME) AND RDB$FIELD_NAME IN (
